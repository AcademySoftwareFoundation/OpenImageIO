00107 inline unsigned int
00108 strhash (const char *s)
00109 {
00110     if (!s) return 0;
00111     unsigned int h=0, g;
00112     while (*s) {
00113         h = (h<<4) + (unsigned char)(*s);
00114         if ((g = (h & 0xf0000000))) {
00115             h ^= g>>24;
00116             h ^= g;
00117         }
00118         ++s;
00119     }
00120     return h;
00121 }
